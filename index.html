import re text = "سایت غیرفعال @lta-seven.vercel.app سایت غیرفعال است" # روش ۱: حذف دقیق همان آدرس cleaned = text.replace("@lta-seven.vercel.app", "").strip() # روش ۲: حذف تمام کاراکترهای انگلیسی، عدد، @ و نقطه و خط تیره cleaned = re.sub(r'[a-zA-Z0-9@.\-/_]+', '', text).strip() # حذف فاصله‌های اضافی cleaned = re.sub(r'\s+', ' ', cleaned) print(cleaned) # خروجی: سایت غیرفعال سایت غیرفعال است
